    #include <iostream>

#define endl "\n"

using namespace std;

const int maxn = 5e5;
const int sqrn = 700;

int a[maxn + 5];
int R[sqrn + 5][sqrn + 5];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t, od, x, y;
    cin >> t;
    while (t--) {
        cin >> od >> x >> y;
        if (od == 1) {
            a[x] += y;
            for (int i = 1; i <= sqrn; i++) {
                R[i][x % i] += y;
            }
        } else {
            if (x <= sqrn) cout << R[x][y] << endl;
            else {
                int ans = 0;
                for (int i = y; i <= maxn; i += x) {
                    ans += a[i];
                }
                cout << ans << endl;
            }
        }
    }

    return 0;
}
